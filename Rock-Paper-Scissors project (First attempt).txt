#include <iostream>
#include <string>
#include <cstdlib>

using namespace std;

enum enChoice { Stone = 1, Paper = 2, Scissors = 3 };
enum enWinner { Player = 1, Computer = 2, Draw = 3 };

int RandomNumber(int From, int To)
{
    int RandNum = rand() % (To - From + 1) + From;

    return RandNum;
}

short ReadNumberOfRounds()
{
    short Rounds = 0;

    cout << "How Many Rounds? 1 to 10?\n";
    cin >> Rounds;

    return Rounds;
}

short EnterPlayerChoice()
{
    short Choice;
    
    do
    {
        cout << "Your Choice : [1]:Stone, [2]:Paper, [3]:Scissors ? ";
        cin>>Choice;

    } while (!(Choice >= 1 && Choice <= 3));

    return Choice;
}

enChoice PlayerChoice()
{
    switch (EnterPlayerChoice())
    {
    case 1:
        return enChoice::Stone;
    case 2:
        return enChoice::Paper;
    case 3:
        return enChoice::Scissors;
    }

}

short GetComputerChoice()
{
    return RandomNumber(1, 3);
}

enChoice ComputerChoice()
{
    switch (GetComputerChoice())
    {
    case 1:
        return enChoice::Stone;
    case 2:
        return enChoice::Paper;
    case 3:
        return enChoice::Scissors;
    }
}

enWinner WhoWin(enChoice Player, enChoice Computer)
{
    if (Player == Computer)
        return enWinner::Draw;
    else if (Player == enChoice::Stone && Computer == enChoice::Paper)
        return enWinner::Computer;
    else if (Player == enChoice::Stone && Computer == enChoice::Scissors)
        return enWinner::Player;
    else if (Player == enChoice::Paper && Computer == enChoice::Stone)
        return enWinner::Player;
    else if (Player == enChoice::Paper && Computer == enChoice::Scissors)
        return enWinner::Computer;
    else if (Player == enChoice::Scissors && Computer == enChoice::Stone)
        return enWinner::Computer;
    else
        return enWinner::Player;
    
}

string ChoiceToString(enChoice Choice)
{
    switch (Choice)
    {
    case enChoice::Stone:
        return "Stone";
    case enChoice::Paper:
        return "Paper";
    case enChoice::Scissors:
        return "Scissors";
    }
}

string WinnerToString(enWinner Winner)
{
    switch (Winner)
    {
    case enWinner::Player:
        return "[Player]";
    case enWinner::Computer:
        return "[Computer]";
    case enWinner::Draw:
        return "[No Winner]";
    }
}

void PrintRoundResult(enWinner Winner, enChoice Player, enChoice Computer)
{
    cout << "Player Choice : " << ChoiceToString(Player) << endl;
    cout << "Computer Choice : " << ChoiceToString(Computer) << endl;
    cout << "Round Winner : " << WinnerToString(Winner);
}

void ScreenColor(enWinner Winner)
{
    if (Winner == enWinner::Player)
        system("color 2f");
    else if (Winner == enWinner::Computer)
        system("color 4f");
    else
        system("color 6f");
}

void LoseBell(enWinner Winner)
{
    if (Winner == enWinner::Computer)
        cout << "\a";
}

void WinsCounter(enWinner Winner, short& PlayerWins, short& ComputerWins, short& Drawtimes)
{
    if (Winner == enWinner::Player)
        PlayerWins++;
    else if (Winner == enWinner::Computer)
        ComputerWins++;
    else
        Drawtimes++;
}

string WhoFinalWin(short PlayerWins, short ComputerWins)
{

    if (PlayerWins > ComputerWins)
        return "Player";
    else if (ComputerWins > PlayerWins)
        return "Computer";
    else
        return "Draw";
}

void GameResult(short Rounds, short PlayerWins, short ComputerWins, short DrawTimes)
{
    cout << "\n\t\t\t_______________________________________________________________________\n\n";
    cout << "\t\t\t\t\t\t+++ Game Over +++\n";
    cout << "\n\t\t\t_______________________________________________________________________\n\n";
    cout << "\t\t\t________________________ [Game Results] _________________________________\n\n";
    cout << "\t\t\tGame Rounds       : " << Rounds << endl;
    cout << "\t\t\tPlayer won times  : " << PlayerWins << endl;
    cout << "\t\t\tComputer won time : " << ComputerWins << endl;
    cout << "\t\t\tDraw times        : " << DrawTimes << endl;
    cout << "\t\t\tFinal Winner      :" << WhoFinalWin(PlayerWins, ComputerWins) << endl;
    cout << "\n\t\t\t_______________________________________________________________________\n";
}

bool PlayAgain()
{
    char Answer;

    cout << "\nDo you want to play again ? (y/n) : ";
    cin >> Answer;

    return (Answer == 'y' || Answer == 'Y');
}

void ResetScreen()
{
    system("cls");
    system("color 0f");
}

void BeginRound()
{
    short Rounds = ReadNumberOfRounds();

    short PlayerWins = 0, ComputerWins = 0, DrawTimes = 0;

    for (int i = 1;i <= Rounds;i++)
    {
        cout << "\nRound [" << i << "] begins :\n\n";

        enChoice Player = PlayerChoice();
        enChoice Computer = ComputerChoice();

        enWinner Winner = WhoWin(Player, Computer);

        ScreenColor(Winner);
        LoseBell(Winner);

        WinsCounter(Winner, PlayerWins, ComputerWins, DrawTimes);

        cout << "\n_______________Round [" << i << "]_______________\n\n";
        PrintRoundResult(Winner, Player, Computer);
        cout << "\n_______________________________________\n";
    }

    GameResult(Rounds, PlayerWins, ComputerWins, DrawTimes);

}

void StartGame()
{
    do
    {
        ResetScreen();
        BeginRound();

    } while (PlayAgain());
}

int main()
{
    srand((unsigned)time(NULL));

    StartGame();

    return 0;

}
